        // Generate all contract files
        function generateAllContracts() {
            const statusDiv = document.getElementById('contractGenerationStatus');
            statusDiv.innerHTML = '
🔄 Generating contract files...
';
            
            const contracts = {};
            
            Object.keys(approvedFreelancers).forEach(freelancerName => {
                const freelancerData = approvedFreelancers[freelancerName];
                const { fileName, contractHTML } = generateContractHTML(freelancerName, freelancerData);
                contracts[fileName] = contractHTML;
            });
            
            // Store contracts for download
            window.generatedContracts = contracts;
            
            statusDiv.innerHTML = `
                
✅ Generated ${Object.keys(contracts).length} contract files:

                
    
                    ${Object.keys(contracts).map(fileName => `
  • ${fileName}
    • `).join('')}
                

            `;
            
            document.getElementById('downloadContractsBtn').style.display = 'block';
        }

        // Download all contract files as a zip
        function downloadContractFiles() {
            if (!window.generatedContracts) {
                alert('Please generate contracts first.');
                return;
            }
            
            // Create a zip file with all contracts
            const zip = new JSZip();
            
            Object.keys(window.generatedContracts).forEach(fileName => {
                zip.file(fileName, window.generatedContracts[fileName]);
            });
            
            zip.generateAsync({type: 'blob'}).then(function(content) {
                const url = URL.createObjectURL(content);
                const a = document.createElement('a');
                a.href = url;
                a.download = 'contracts.zip';
                document.body.appendChild(a);
                a.click();
                document.body.removeChild(a);
                URL.revokeObjectURL(url);
            });
        }

        // Initialize on page load
        document.addEventListener('DOMContentLoaded', function() {
            console.log('DOM loaded, initializing...');

        });
        
        // Fallback initialization
        if (document.readyState === 'loading') {
            console.log('Document still loading, waiting for DOMContentLoaded...');
        } else {
